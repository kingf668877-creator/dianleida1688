41let aiEnabled = true;
42function toggleAI() {
43  aiEnabled = !aiEnabled;
44  const toggle = document.getElementById('aiToggle');
45  const aiCard = document.getElementById('aiCard');
46  if (aiEnabled) {
47    toggle.classList.remove('off');
48    toggle.querySelector('.ai-toggle-check').textContent = '✓';
49    aiCard.style.display = '';
50  } else {
51    toggle.classList.add('off');
52    toggle.querySelector('.ai-toggle-check').textContent = '';
53    aiCard.style.display = 'none';
54  }
55}
